<body>
    <div id="canvas-container"></div>
    <div id="ui-overlay">Træk for at rotere • Scroll for at zoome • Det episke sværd</div>

    <!-- Henter Three.js og OrbitControls via CDN -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/three@0.128.0/examples/js/controls/OrbitControls.js"></script>

    <script>
        let scene, camera, renderer, controls;
        let swordGroup; // Gør swordGroup global, så vi kan animere den

        window.onload = function() {
            init();
            animate();
        };

        function init() {
            // 1. Opsætning af scene
            const container = document.getElementById('canvas-container');
            scene = new THREE.Scene();
            scene.background = new THREE.Color('#1a1a1a');

            // 2. Kamera
            camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 100);
            camera.position.set(0, 0, 30);

            // 3. Renderer med glow/post-processing setup (basal version via emissive)
            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.shadowMap.enabled = true;
            renderer.shadowMap.type = THREE.PCFSoftShadowMap;
            container.appendChild(renderer.domElement);

            // 4. Belysning
            const ambientLight = new THREE.AmbientLight(0xffffff, 0.3);
            scene.add(ambientLight);

            const dirLight = new THREE.DirectionalLight(0xffffff, 0.8);
            dirLight.position.set(5, 10, 7);
            dirLight.castShadow = true;
            scene.add(dirLight);

            const fillLight = new THREE.DirectionalLight(0x4444ff, 0.5); // Koldt blåt lys fra siden som kontrast
            fillLight.position.set(-5, 0, -5);
            scene.add(fillLight);

            // Tilføj et magisk rødt lys, der udsender fra sværdet (Gems)
            const gemLight = new THREE.PointLight(0xff0055, 1.5, 15);
            gemLight.position.set(0, -0.2, 0.5);
            scene.add(gemLight);

            // 5. Oprettelse af 3D Meshes (Bygger det vilde sværd)
            swordGroup = new THREE.Group();

            // --- MATERIALER ---
            const metalMaterial = new THREE.MeshStandardMaterial({ 
                color: '#4a4e59', 
                roughness: 0.3,
                metalness: 0.9,
                flatShading: false
            });

            const darkMetalMaterial = new THREE.MeshStandardMaterial({ 
                color: '#111', 
                roughness: 0.5,
                metalness: 0.8,
            });

            const leatherMaterial = new THREE.MeshStandardMaterial({ 
                color: '#1a0d0a', // Meget mørkt læder
                roughness: 0.9,
                metalness: 0.1
            });

            // Gennemkrystet, glødende ædelstens-materiale (Ruby)
            const gemMaterial = new THREE.MeshPhysicalMaterial({
                color: 0xff0055,
                metalness: 0.2,
                roughness: 0.1,
                transmission: 0.9, // Glas-effekt
                thickness: 0.5,
                emissive: 0x990033, // Gløder indefra
                emissiveIntensity: 0.6,
                clearcoat: 1.0,
                clearcoatRoughness: 0.1
            });

            // --- KLINGE (Blade) ---
            const bladeShape = new THREE.Shape();
            bladeShape.moveTo(0, 0); 
            bladeShape.lineTo(0.9, 0); // Gjort bredere i bunden for et mere brutalt look
            bladeShape.lineTo(0.6, 9); 
            bladeShape.lineTo(0, 12.5); // Længere, skarpere spids
            bladeShape.lineTo(-0.6, 9); 
            bladeShape.lineTo(-0.9, 0); 
            bladeShape.lineTo(0, 0); 

            const extrudeSettings = {
                steps: 2,
                depth: 0.05, 
                bevelEnabled: true,
                bevelThickness: 0.15, 
                bevelSize: 0.1,
                bevelOffset: 0,
                bevelSegments: 3
            };

            const bladeGeo = new THREE.ExtrudeGeometry(bladeShape, extrudeSettings);
            bladeGeo.center(); 
            const bladeMesh = new THREE.Mesh(bladeGeo, metalMaterial);
            bladeMesh.position.y = 5.5; 
            bladeMesh.castShadow = true;
            swordGroup.add(bladeMesh);

            // En mørk "blodrille" (fuller) i midten af klingen
            const fullerGeo = new THREE.BoxGeometry(0.15, 8, 0.4);
            const fullerMesh = new THREE.Mesh(fullerGeo, darkMetalMaterial);
            fullerMesh.position.y = 4.5;
            swordGroup.add(fullerMesh);

            // --- FANTASY PARERPLADE (Crossguard) ---
            const crossguardGroup = new THREE.Group();
            crossguardGroup.position.y = -0.2;

            // Centrums-blok
            const cgCenterGeo = new THREE.BoxGeometry(1.5, 1.2, 0.8);
            const cgCenterMesh = new THREE.Mesh(cgCenterGeo, darkMetalMaterial);
            crossguardGroup.add(cgCenterMesh);

            // Vinger (Aggressive pigge/horn)
            const wingGeo = new THREE.ConeGeometry(0.3, 3, 4); // 4-sidet kegle
            
            const rightWing = new THREE.Mesh(wingGeo, metalMaterial);
            rightWing.position.set(1.5, 0.5, 0);
            rightWing.rotation.z = -Math.PI / 3; // Peger opad/udad
            crossguardGroup.add(rightWing);

            const leftWing = new THREE.Mesh(wingGeo, metalMaterial);
            leftWing.position.set(-1.5, 0.5, 0);
            leftWing.rotation.z = Math.PI / 3;
            crossguardGroup.add(leftWing);

            swordGroup.add(crossguardGroup);

            // --- DEN STORE ÆDELSTEN (Crossguard Gem) ---
            const mainGemGeo = new THREE.OctahedronGeometry(0.6, 0); // Krystalform
            const mainGem = new THREE.Mesh(mainGemGeo, gemMaterial);
            mainGem.position.y = -0.2;
            mainGem.position.z = 0.4; // Stikker ud fra centeret
            mainGem.scale.set(1, 1.2, 0.6); // Fladtrykt lidt, strakt på y
            swordGroup.add(mainGem);

            // Og en tilsvarende på bagsiden
            const mainGemBack = mainGem.clone();
            mainGemBack.position.z = -0.4;
            swordGroup.add(mainGemBack);

            // --- HÅNDTAG (Grip) ---
            const gripGeo = new THREE.CylinderGeometry(0.35, 0.3, 3.5, 32);
            const gripMesh = new THREE.Mesh(gripGeo, leatherMaterial);
            gripMesh.position.y = -2.1;
            gripMesh.castShadow = true;
            swordGroup.add(gripMesh);

            const bandGeo = new THREE.TorusGeometry(0.38, 0.05, 16, 32);
            const topBand = new THREE.Mesh(bandGeo, metalMaterial);
            topBand.position.y = -0.4;
            topBand.rotation.x = Math.PI / 2;
            swordGroup.add(topBand);

            // --- KNAP MED ÆDELSTEN (Pommel) ---
            const pommelGroup = new THREE.Group();
            pommelGroup.position.y = -4.0;

            const pommelBaseGeo = new THREE.CylinderGeometry(0.5, 0.2, 0.8, 4); // Skarp 4-kantet klo
            const pommelBaseMesh = new THREE.Mesh(pommelBaseGeo, darkMetalMaterial);
            pommelBaseMesh.rotation.y = Math.PI / 4;
            pommelGroup.add(pommelBaseMesh);

            // Den lille ædelsten i bunden
            const bottomGemGeo = new THREE.OctahedronGeometry(0.4, 0);
            const bottomGem = new THREE.Mesh(bottomGemGeo, gemMaterial);
            bottomGem.position.y = -0.4;
            pommelGroup.add(bottomGem);

            swordGroup.add(pommelGroup);

            // Tilføj hele sværdet til scenen
            scene.add(swordGroup);

            // Oplys sværdet let med lyset
            gemLight.target = mainGem;

            // 6. Controls
            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.dampingFactor = 0.05;
            controls.minDistance = 10;
            controls.maxDistance = 60;

            window.addEventListener('resize', onWindowResize, false);
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        function animate() {
            requestAnimationFrame(animate);
            controls.update(); 

            // Magisk svævende og let roterende animation
            const time = Date.now() * 0.001;
            if (swordGroup) {
                swordGroup.position.y = Math.sin(time * 2) * 0.3; // Bobler op og ned
                swordGroup.rotation.y = Math.sin(time * 0.5) * 0.2; // Drejer let frem og tilbage
            }

            renderer.render(scene, camera);
        }
    </script>
